% ---------- LaTeX-escape helper (Jinja macro) ----------
{% macro esc(s) -%}
{{-  s
     | replace('\\', '\textbackslash{}')
     | replace('&',  '\&')
     | replace('%',  '\%')
     | replace('$',  '\$')
     | replace('#',  '\#')
     | replace('_',  '\_')
     | replace('{',  '\{')
     | replace('}',  '\}')
     | replace('~',  '\textasciitilde{}')
     | replace('^',  '\textasciicircum{}')
-}}
{%- endmacro %}
